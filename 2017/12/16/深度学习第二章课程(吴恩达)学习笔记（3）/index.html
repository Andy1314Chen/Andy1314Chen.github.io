import numpy as np

# sigmoid函数
def sigmoid(z):
    s = 1 / (1 + np.exp(-z))
    return s

# initializing parameters
def initialize_with_zeros(dim):
    W = np.zeros((dim, 1))
    b = 0
    assert(W.shape == (dim, 1))
    assert(isinstance(b, float) or isinstance(b, int))
    return W,b

# 传播
def propagate(w, b, X, Y):
    m = X.shape[1] # 样本数 X:(nx特征数, m样本数) Y: (1,m)
    # FORWARD PROPAGATION
    A = sigmoid(np.dot(w.T, X)+b)
    cost = -1/m*np.sum(Y*np.log(A)+(1-Y)*np.log(1-A))

    # BACKWARD PROPAGATION
    dw = 1/m * np.dot(X, (A-Y).T)
    db = 1/m * np.sum(A-Y)

    assert(dw.shape == w.shape)
    assert(db.dtype == float)

    cost = np.squeeze(cost)
    assert(cost.shape == ())

    grads = {'dw':dw,
             'db':db}
    return grads, cost

# GRADED FUNCTION: optimize
def optimize(w, b, X, Y, num_iterations, learning_rate, print_cost=False):
    costs = []
    for i in range(num_iterations):
        # 正向传播计算cost, 反向传播计算grad
        grads, cost = propagate(w, b, X, Y)

        dw = grads['dw']
        db = grads['db']
        # 负梯度方向下降
        w = w - learning_rate*dw
        b = b - learning_rate*db

        if i%100 == 0:
            costs.append(cost)

        if print_cost and i%100 == 0:
            print("Cost after iteration %i: %f" % (i, cost))
    params = {'w':w,
              'b':b}
    grads = {'dw':dw,
             'db':db}
    return params, grads, costs

# 预测    
def predict(w, b, X):
    m = X.shape[1]
    Y_prediction = np.zeros((1,m))
    w = w.reshape(X.shape[0], 1)

    A = sigmoid(np.dot(w.T, X)+b)

    for i in range(A.shape[1]):
        if A[0, i] <= 0.5:
            Y_prediction[0, i] = 0
        else:
            Y_prediction[0, i] = 1
    assert(Y_prediction.shape == (1, m))
    return Y_prediction

# 迭代100次，梯度下降法训练参数
#params, grads, costs = optimize(w, b, X, Y, num_iterations=100, learning_rate=0.009,
#                                print_cost=True)


def model(X_train, Y_train, X_test, Y_test, num_iterations = 2000, learning_rate=0.5,
          print_cost = False):
    # 初始化参数
    w, b = initialize_with_zeros(X_train.shape[0])
    # 梯度下降法优化
    parameters, grads, costs = optimize(w, b, X_train, Y_train,
                                        num_iterations, learning_rate, print_cost)
    # 得到参数
    w = parameters['w']
    b = parameters['b']

    # 预测
    Y_prediction_test = predict(w, b, X_test)
    Y_prediction_train = predict(w,b, X_train)

    # 误差
    print("train accuracy: {} %".format(100 - np.mean(np.abs(Y_prediction_train - Y_train)) * 100))
    print("test accuracy: {} %".format(100 - np.mean(np.abs(Y_prediction_test - Y_test)) * 100))

    # 
    d = {"costs": costs,
         "Y_prediction_test": Y_prediction_test, 
         "Y_prediction_train" : Y_prediction_train, 
         "w" : w, 
         "b" : b,
         "learning_rate" : learning_rate,
         "num_iterations": num_iterations}

    return d



from sklearn.datasets import load_iris
from sklearn.linear_model import LogisticRegression

# iris数据集
iris = load_iris()
# 样本
samples = iris.data
samples = samples[:-50]
# 标签
target = iris.target
target = target[:-50]

clf_lr = LogisticRegression()
clf_lr.fit(samples, target)

y_train = clf_lr.predict(samples)
print(sum(y_train == target)/len(target))

# 上面的model输入，X(特征数，样本数)，Y(1,样本数)与sklearn库不太一样
samples2 = samples.T
target2 = target.reshape((1, len(target)))
model(samples2, target2, samples2, target2, num_iterations=1000, learning_rate=0.001)
